==== Convert Hex to Decimal ====

<code cpp>
#include <sstream>
#include <iostream>

int main() {
	int x;
	std::stringstream ss;
	ss << std::hex << "ff";
	ss >> x;
	std::cout << x << std::endl;
}
</code>
